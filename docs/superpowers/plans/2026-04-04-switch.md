> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `/switch <目标路径>` command that stops the current bridge, copies its config to the target project, starts the target bridge, then stops the current one — enabling multi-project context switching.

**Architecture:** `switcher.py` contains the pure switching logic (no async, no Feishu SDK). It's called from both the CLI (`main.py switch` subcommand) and the Feishu message handler (`_handle_switch` in `message_handler.py`). The switcher raises typed exceptions on failure; callers handle the error display.

**Tech Stack:** Python stdlib (`subprocess`, `signal`, `yaml`, `pathlib`), no new dependencies.

```python
# cc_feishu_bridge/switcher.py
"""Project switcher: stop current bridge, copy config to target, start target bridge."""
from __future__ import annotations

import os
import signal
import subprocess
import time
from pathlib import Path
from dataclasses import dataclass
from typing import Literal

import yaml


class SwitchError(Exception):
    """Base exception for switcher errors."""
    pass


class NotInitializedError(SwitchError):
    """Target project has not been initialized with a Feishu bot config."""
    pass


class TargetStopError(SwitchError):
    """Failed to stop target bridge."""
    pass


class CurrentStopError(SwitchError):
    """Failed to stop current bridge."""
    pass


class StartupTimeoutError(SwitchError):
    """Target bridge did not start within the timeout."""
    pass


@dataclass
class SwitchResult:
    success: bool
    target_path: str
    target_pid: int | None = None
    error_step: str | None = None
    error_message: str | None = None


def _is_initialized(cc_dir: str) -> bool:
    """Return True if cc_dir/config.yaml has valid feishu credentials."""
    cfg_path = os.path.join(cc_dir, "config.yaml")
    if not os.path.exists(cfg_path):
        return False
    try:
        with open(cfg_path) as f:
            cfg = yaml.safe_load(f)
        feishu = cfg.get("feishu", {})
        return bool(feishu.get("app_id")) and bool(feishu.get("app_secret"))
    except Exception:
        return False


def _resolve_cc_dir(path: str) -> str:
    """Return .cc-feishu-bridge dir for a given project path."""
    return os.path.join(os.path.abspath(path), ".cc-feishu-bridge")


def _stop_bridge(pid_file: str, timeout: float = 5.0) -> bool:
    """Stop a bridge by PID file. Returns True if stopped, False if not running."""
    if not os.path.exists(pid_file):
        return True  # not running, nothing to do
    try:
        pid = int(Path(pid_file).read_text().strip())
    except (ValueError, OSError):
        return True
    try:
        os.kill(pid, 0)
    except OSError:
        return True  # already dead
    try:
        os.kill(pid, signal.SIGTERM)
        # Wait for it to die
        start = time.time()
        while time.time() - start < timeout:
            try:
                os.kill(pid, 0)
            except OSError:
                return True
            time.sleep(0.2)
        os.kill(pid, signal.SIGKILL)
        return True
    except OSError:
        return True


def _start_bridge(target_cc: str) -> int:
    """Start bridge in target_cc directory. Returns PID. Raises StartupTimeoutError on failure."""
    import sys
    # Use the same Python interpreter running right now
    python = sys.executable
    proc = subprocess.Popen(
        [python, "-m", "cc_feishu_bridge.main"],
        cwd=os.path.dirname(target_cc),
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
        start_new_session=True,
    )
    # Wait for pid file to appear (max 8s)
    pid_file = os.path.join(target_cc, "cc-feishu-bridge.pid")
    for _ in range(40):
        if os.path.exists(pid_file):
            try:
                return int(Path(pid_file).read_text().strip())
            except (ValueError, OSError):
                pass
        time.sleep(0.2)
    raise StartupTimeoutError(f"Bridge in {target_cc} did not start in time")


def switch_to(target_path: str) -> SwitchResult:
    """
    Execute the full project switch:
      1. Check target initialized
      2. Stop target bridge (if running)
      3. Copy & fix config.yaml to target
      4. Start target bridge
      5. Stop current bridge
    Returns SwitchResult. Errors do not roll back.
    """
    target_cc = _resolve_cc_dir(target_path)

    # Step 1: initialized check
    if not _is_initialized(target_cc):
        return SwitchResult(
            success=False,
            target_path=target_path,
            error_step="check_initialized",
            error_message="目标项目未初始化。请先运行 cc-feishu-bridge install 完成初始化。",
        )

    # Step 2: stop target bridge if running
    target_pid_file = os.path.join(target_cc, "cc-feishu-bridge.pid")
    _stop_bridge(target_pid_file)

    # Step 3: copy and fix config.yaml
    current_cc = os.path.join(os.getcwd(), ".cc-feishu-bridge")
    current_cfg_path = os.path.join(current_cc, "config.yaml")
    with open(current_cfg_path) as f:
        cfg = yaml.safe_load(f)

    # Rewrite db_path to target
    cfg["storage"]["db_path"] = os.path.join(target_cc, "sessions.db")

    os.makedirs(target_cc, exist_ok=True)
    with open(os.path.join(target_cc, "config.yaml"), "w") as f:
        yaml.dump(cfg, f, default_flow_style=False, allow_unicode=True)

    # Step 4: start target bridge
    try:
        new_pid = _start_bridge(target_cc)
    except StartupTimeoutError as e:
        return SwitchResult(
            success=False,
            target_path=target_path,
            error_step="start_target",
            error_message=str(e),
        )

    # Step 5: stop current bridge
    current_pid_file = os.path.join(current_cc, "cc-feishu-bridge.pid")
    _stop_bridge(current_pid_file)

    return SwitchResult(success=True, target_path=target_path, target_pid=new_pid)
```

```python
    async def _handle_switch(self, message: IncomingMessage) -> HandlerResult:
        """Handle /switch <target-path> command."""
        from cc_feishu_bridge.switcher import switch_to, SwitchResult
        import os

        parts = message.content.split(maxsplit=1)
        if len(parts) < 2:
            return HandlerResult(
                success=True,
                response_text="用法: /switch <目标项目路径>\n例: /switch /Users/x/my-project",
            )

        raw_path = parts[1].strip()
        # Resolve relative paths against current cwd
        if raw_path.startswith("/") or raw_path.startswith("~"):
            target = os.path.expanduser(raw_path)
        else:
            target = os.path.abspath(raw_path)

        await self.feishu.add_typing_reaction(message.message_id)

        result: SwitchResult = switch_to(target)

        if result.success:
            card_md = (
                f"## ✅ 已切换到新项目\n\n"
                f"| 项目 | 值 |\n"
                f"|------|-----|\n"
                f"| 目标目录 | `{target}` |\n"
                f"| 新 Bridge PID | `{result.target_pid}` |\n\n"
                f"> 飞书消息流已切换，请在目标项目下继续对话。\n"
                f"> 返回时执行 `/switch <当前路径>` 即可。"
            )
            await self._safe_send(message.chat_id, message.message_id, card_md)
            return HandlerResult(success=True, response_text="")
        else:
            error_text = f"❌ 切换失败\n\n**失败步骤**: {result.error_step}\n\n**原因**: {result.error_message}"
            await self._safe_send(message.chat_id, message.message_id, error_text)
            return HandlerResult(success=True, response_text="")
```
